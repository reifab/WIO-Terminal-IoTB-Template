36template<class type>
37struct RawImage{
38 type * ptr(){
39 return (type *)(this + 1);
40 }
41 type get(int16_t x, int16_t y){
42 return this->ptr()[y * width() + x];
43 }
44 void draw(size_t x = 0, size_t y = 0){
45 tft.pushImage(x, y, width(), height(), ptr());
46 }
47 void release(){
48 delete [] this;
49 }
50 int16_t width(){ return _width; }
51 int16_t height(){ return _height; }

60template<class type>
61RawImage<type> * newImage(const char * path){
62 typedef RawImage<type> raw;
63 File f = SD.open(path, FILE_READ);
64 if (!f){
65 return nullptr;
66 }
67 int32_t size = f.size();
68 raw * mem = (raw *)new uint8_t[size];
69 if (mem == nullptr){
70 return nullptr;
71 }
72 f.read(mem, size);
73 f.close();
74 return mem;
75}
